Angular 2 family

@stackline/angular-multiselect-dropdown

This Angular 2 documentation uses the exact examples from the final local test app: Angular 2.4.10 with @stackline/angular-multiselect-dropdown@2.0.4.

The matrix runs the same twelve scenarios in classic and material. The Angular 2 test validates the legacy <angular2-multiselect> selector while the maintained package also supports <angular-multiselect>.

Published package 2.0.4 for Angular 2.x
Peer range >=2.0.0 <3.0.0
Skins theme: 'classic' and theme: 'material'
Examples 12 scenarios per skin, 24 documented cases total.

Setup used by the test

Install 
npm install @stackline/angular-multiselect-dropdown@2.0.4
Register the module 
import { FormsModule } from '@angular/forms';
import { AngularMultiSelectModule } from '@stackline/angular-multiselect-dropdown';

@NgModule({
  imports: [BrowserModule, FormsModule, AngularMultiSelectModule]
})
export class AppModule {}
Render the Angular 2 selector 
<angular2-multiselect
  [data]="countries"
  [(ngModel)]="classicBasic"
  [settings]="classicBasicSettings"
  (onSelect)="record('classic basic select', $event)"
  (onDeSelect)="record('classic basic deselect', $event)"
  (onSelectAll)="record('classic basic selectAll', $event)"
  (onDeSelectAll)="record('classic basic deselectAll', $event)">
</angular2-multiselect>

Shared settings factory 
makeSettings(theme: string, text: string, options: any) {
  var settings: any = {
    singleSelection: false,
    text: text,
    selectAllText: 'Selecionar todos',
    unSelectAllText: 'Limpar todos',
    enableCheckAll: true,
    enableSearchFilter: true,
    searchPlaceholderText: 'Buscar',
    badgeShowLimit: 4,
    maxHeight: 260,
    showCheckbox: true,
    noDataLabel: 'Sem dados',
    theme: theme
  };

  for (var key in options) {
    if (options.hasOwnProperty(key)) {
      settings[key] = options[key];
    }
  }

  return settings;
}

</angular2-multiselect>
